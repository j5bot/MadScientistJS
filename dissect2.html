"Public" and "Private" Methods

With the proper use of function scope, we can create "inner" or "private" methods.

function bar () { function foo () { alert("foo"); } } alert(typeof foo); // undefined alert(typeof bar.foo); // undefined

"Public" and "Private" Methods

We can also create public methods (in quite a few ways)

function bar () { this.foo = function () { alert("foo"); }; // note that this is an assignment statement, not a declaration, so it ends with a semi-colon } bar.bad = function() { alert("bad!"); }; // functions are objects, so you can assign other functions as properties bar["baz"] = function () { alert("baz!"); }; // another way to create an assignment ... alert("bar.foo is: " + typeof(bar.foo)); alert("bar.bad is: " + typeof(bar.bad)); alert("bar.baz is: " + typeof(bar.baz)); var b = new bar(); alert("b.foo is: " + typeof(b.foo));
